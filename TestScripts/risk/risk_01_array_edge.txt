// 배열 경계 동작 검증
// - 음수 인덱스: arr[-1]
// - 소수 크기:   Array(3.7)
// - 크기 0:      Array(0)
// - 인자 초과:   Func foo(a){} foo(1,2)
// ──────────────────────────────────────────────────────────────────────────

// [1] 소수 크기 Array(3.7) → 크기 3으로 처리되는가 vs 오류인가
print "=== Array(3.7) ===";
var arr1 = Array(3.7);
arr1[0] = 10;
print arr1[0];   // expect: 10  (size_t 변환 시 3으로 잘림)
