Func add(a, b) {
    return a + b;
}
print add(3, 7);        // 10
print add(100, 200);    // 300

Func factorial(n) {
    if (n <= 1) return 1;
    return n * factorial(n - 1);
}
print factorial(1);     // 1
print factorial(5);     // 120
print factorial(10);    // 3628800

Func fib(n) {
    if (n <= 1) return n;
    return fib(n - 1) + fib(n - 2);
}
print fib(0);           // 0
print fib(1);           // 1
print fib(10);          // 55

Func counter(start, step, count) {
    var cur = start;
    for (var i = 0; i < count; i = i + 1) {
        cur = cur + step;
    }
    return cur;
}
print counter(0, 5, 4); // 20
print counter(10, 3, 3);// 19
