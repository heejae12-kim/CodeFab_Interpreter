// 출처: 요구사항 gist Section 1-1
// 실행: x64/Debug/CodeFab_Interpreter.exe run TestScripts/test_06_precedence.txt
// ──────────────────────────────────────────────────────────────────────────

// --- 산술 우선순위 (곱셈·나눗셈 > 덧셈·뺄셈) ---
print 1 + 2 * 3;        // expect: 7
print (1 + 2) * 3;      // expect: 9

// --- 좌결합 (left-associative) ---
print 10 - 4 - 3;       // expect: 3
print 8 / 2 / 2;        // expect: 2

// --- 단항 마이너스 + 이항 덧셈 ---
print -3 + 2;            // expect: -1

// --- 비교 ---
print 1 < 2;             // expect: true
print 3 > 5;             // expect: false

// --- 문자열 연결 ---
print "Hello, " + "CodeFab!";   // expect: Hello, CodeFab!

// --- 숫자 출력 포맷 (정수값이면 소수점 없이 출력) ---
print 5;                 // expect: 5
print 5.0;               // expect: 5
print 3.14;              // expect: 3.14

// --- boolean 리터럴 출력 ---
print true;              // expect: true
print false;             // expect: false
