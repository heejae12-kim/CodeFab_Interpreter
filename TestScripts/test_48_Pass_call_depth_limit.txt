// MAX_CALL_DEPTH(=30) 이내의 재귀 호출이 정상 동작함을 검증
// ──────────────────────────────────────────────────────────────────────────

Func deep(n) {
    if (n <= 0) return "done";
    return deep(n - 1);
}

print deep(1);    // expect: done
print deep(10);   // expect: done
print deep(20);   // expect: done
print deep(29);   // expect: done  (한계 직전)
