// MAX_CALL_DEPTH(=30) 초과 재귀 시 RuntimeError 발생함을 검증
// 기대 오류: "Maximum call stack size exceeded."
// ──────────────────────────────────────────────────────────────────────────

Func deep(n) {
    if (n <= 0) return "done";
    return deep(n - 1);
}

print deep(31);
