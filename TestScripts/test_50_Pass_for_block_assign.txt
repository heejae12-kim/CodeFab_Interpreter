// for 루프 내부 블록에서 변수 할당이 정상 동작함을 검증
// 기대 출력: 0 / 1 / 3 / 3
// ──────────────────────────────────────────────────────────────────────────

var x = 0;
for (var i = 0; i < 3; i = i + 1) {
    x = x + i;
    print x;
}
print x;
